def segmentize(image, seg_func):
    
    cols = image.shape[0]
    rows = image.shape[1]
    
    new_image = np.zeros(image.shape, dtype=np.uint8)   
        
    for i in range(cols):
        for j in range(rows):
            new_image[i,j] = np.uint8( seg_func[image[i,j]] )
    return new_image
